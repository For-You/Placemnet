{
    int pivot=a[low],i=low,j=high;
    while(i< j)
    {
    while(a[i]<=pivot&&i< j)
        i++;
    while(a[j]>pivot)
        j--;
    if(i< j)
    {
        int temp=a[i];
        a[i]=a[j];
        a[j]=temp;
    }
  }
  a[low]=a[j];
  a[j]=pivot;
  return j;
}
